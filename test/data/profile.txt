      var activityChart;
      var chartData;
      var chartDataSeriesArray = [];
      var highchartsOptions = {};
      var xUnits = null;

      activityInfoCallbacks.push(function(activityInfo)
      {
         if (activityInfo == null) return;

         // Render activity chart
         if (activityInfo.chartData)
         {
            chartData = activityInfo.chartData;

            for (var seriesId in chartData.series)
            {
               chartData.series[seriesId].visible = true;
               if (xUnits == null) xUnits = chartData.series[seriesId].xUnits;
            }
            
            // Default to showing pace for running and walking activities
            if (((activityInfo.activityType == "RUN")  ||
                 (activityInfo.activityType == "WALK") ||
                 (activityInfo.activityType == "HIKE")) && chartData.series["SPEED"])
            {
               chartData.series["SPEED"].visible = false;
            }
            else if (chartData.series["PACE"])
            {
               chartData.series["PACE"].visible = false;
            }
            
            renderActivityChart();
         }
      });

      function renderActivityChart()
      {
         highchartsOptions.chart =
         {
            renderTo: "activityChartContainer",
            margin: [0, 20, 30, 20],

            alignTicks: false

            itemStyle: {
               margin: "0px 0px 0px 100px"
            }

         highchartsOptions.legend = {
            borderWidth: 0,
            style: {
               left: "auto",
               right: "80px",
               top: "-15px",
               bottom: "auto"
            },
            itemHiddenStyle: {
               color: "#888"
            }
         };
         
         highchartsOptions.series = [];
         var seriesIndex = 0;
         var seriesToAdd = chartData.series["ELEVATION"];

         if (chartData.series["ELEVATION"])
         {
            chartDataSeriesArray[seriesIndex] = seriesToAdd;
            
            highchartsOptions.series[seriesIndex++] = {
               id:   seriesToAdd.id,
               name: seriesToAdd.label,
               data: seriesToAdd.data,
               type: "area",
               color: "rgba(22, 137, 13, 1)",

                     [0, "rgba(62, 177, 53, 0.75)"],

                  ]
               },
               //fillOpacity: 0.65,
               yAxis: 0,
               visible: seriesToAdd.visible
            };
         }

         if (chartData.series["PACE"])
         {
            seriesToAdd = chartData.series["PACE"];
            chartDataSeriesArray[seriesIndex] = seriesToAdd;
            
            highchartsOptions.series[seriesIndex++] = {
               id:   seriesToAdd.id,
               name: seriesToAdd.label,
               data: seriesToAdd.data,
               type: "area",
               color: "#332AC2",
               fillOpacity: 0.25,

            chartDataSeriesArray[seriesIndex] = seriesToAdd;
            
            highchartsOptions.series[seriesIndex++] = {
               id:   seriesToAdd.id,
               name: seriesToAdd.label,
               data: seriesToAdd.data,
               type: "line",
               color: "#AA3333",
               yAxis: 3,
               visible: seriesToAdd.visible
            };
         }
         
         highchartsOptions.xAxis =
         {
            title: {
               enabled: false
            },
            labels: {
                formatter: function() {return (this.value == 0) ? this.value : this.value + " " + xUnits;}
            },
            lineColor: "#99A9B9"
         };
         
         highchartsOptions.yAxis =
         [
            {  // ELEVATION
               gridLineWidth: 0,
               title: {enabled: false, text: ""},
               labels: {enabled: false},
               startOnTick: false,

                      + "Distance: " + Highcharts.numberFormat(this.x, 2) + " "
                      + chartDataSeriesArray[this.series.index].xUnits + timeStr;

         };

         if (activityChart)

            activityChart.destroy();

         activityChart = new Highcharts.Chart(highchartsOptions);
